# Robust session management with error handling
class ResilientAgentCoreClient:
    def __init__(self):
        self.max_retries = 3
        self.failure_count = 0
        self.circuit_breaker_threshold = 0.5  # 50%
    
    def invoke_with_retry(self, session_id, payload):
        for attempt in range(self.max_retries):
            try:
                response = self.client.invoke(
                    session_id=session_id,
                    payload=payload
                )
                self.failure_count = 0  # Reset on success
                return response
            except SessionExpiredException:
                # Session expired - create new one
                session_id = self.session_manager\
                    .create_session(self.runtime_id)
            except Exception as e:
                self.failure_count += 1
                if self._circuit_open():
                    raise CircuitBreakerOpenException()
                wait_time = 2 ** attempt  # Exponential backoff
                time.sleep(wait_time)
